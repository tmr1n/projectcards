ШАГ 4 (потом): Server Actions — реальные запросы к серверу
───────────────────────────────────────────────────────
Файл: src/server-actions/auth.actions.ts

'use server'
export async function loginAction(data: { email: string, password: string }) {
  const res = await fetch('https://your-api/auth/login', {
    method: 'POST',
    body: JSON.stringify(data)
  })
  if (!res.ok) throw new Error('Неверный логин или пароль')
  return res.json() // { user, accessToken }
}

Затем в authStore.ts замени заглушку:
  const response = await loginAction({ email, password })
  set({ user: response.user, accessToken: response.accessToken, ... })
