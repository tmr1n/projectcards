useInputStyles → CVA
src/hooks/useInputStyles.ts — хук, который просто возвращает строку className. Это антипаттерн для простой стилизации. Лучше CVA (как уже используется в кнопках):

})
forwardRef с React 19
В React 19 forwardRef больше не нужен — ref передаётся как обычный prop. InputComponent.tsx использует устаревший паттерн.

// src/server-actions/auth.actions.ts
'use server'
import { redirect } from 'next/navigation'

export async function loginAction(data: LoginFormData) {
  const res = await fetch('/api/auth/login', { method: 'POST', body: JSON.stringify(data) })
  if (!res.ok) return { error: 'Неверный логин или пароль' }
  redirect('/dashboard')
}
